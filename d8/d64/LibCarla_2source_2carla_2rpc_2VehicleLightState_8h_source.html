24 /// Defines the physical appearance of a vehicle whitch is obtained
25 /// by the sensors.
26 class VehicleLightState {
27 public:
28
29 using flag_type = uint32_t;
30
31 /// Can be used as flags
32 enum class LightState : flag_type {
33 None = 0,
34 Position = 0x1,
35 LowBeam = 0x1 << 1,
36 HighBeam = 0x1 << 2,
37 Brake = 0x1 << 3,
38 RightBlinker = 0x1 << 4,
39 LeftBlinker = 0x1 << 5,
40 Reverse = 0x1 << 6,
41 Fog = 0x1 << 7,
42 Interior = 0x1 << 8,
43 Special1 = 0x1 << 9, // E.g: sirens
44 Special2 = 0x1 << 10,
45 All = 0xFFFFFFFF
46 };
47
48 VehicleLightState() = default;
49
50 VehicleLightState(LightState light_state)
51 : light_state(static_cast<flag_type>(light_state)) {}
52
53 VehicleLightState(flag_type light_state)
54 : light_state(light_state) {}
55
56#ifdef LIBCARLA_INCLUDED_FROM_UE4
57
58 VehicleLightState(const FVehicleLightState &InLightState) {
59 light_state = static_cast<flag_type>(LightState::None);
60 if (InLightState.Position) SET_FLAG(light_state, LightState::Position);
61 if (InLightState.LowBeam) SET_FLAG(light_state, LightState::LowBeam);
62 if (InLightState.HighBeam) SET_FLAG(light_state, LightState::HighBeam);
63 if (InLightState.Brake) SET_FLAG(light_state, LightState::Brake);
64 if (InLightState.RightBlinker) SET_FLAG(light_state, LightState::RightBlinker);
65 if (InLightState.LeftBlinker) SET_FLAG(light_state, LightState::LeftBlinker);
66 if (InLightState.Reverse) SET_FLAG(light_state, LightState::Reverse);
67 if (InLightState.Fog) SET_FLAG(light_state, LightState::Fog);
68 if (InLightState.Interior) SET_FLAG(light_state, LightState::Interior);
69 if (InLightState.Special1) SET_FLAG(light_state, LightState::Special1);
70 if (InLightState.Special2) SET_FLAG(light_state, LightState::Special2);
71 }
72
73 operator FVehicleLightState() const {
74 FVehicleLightState Lights; // By default all False
75 if (FLAG_ENABLED(light_state, LightState::Position)) Lights.Position = true;
76 if (FLAG_ENABLED(light_state, LightState::LowBeam)) Lights.LowBeam = true;
77 if (FLAG_ENABLED(light_state, LightState::HighBeam)) Lights.HighBeam = true;
78 if (FLAG_ENABLED(light_state, LightState::Brake)) Lights.Brake = true;
79 if (FLAG_ENABLED(light_state, LightState::RightBlinker)) Lights.RightBlinker = true;
80 if (FLAG_ENABLED(light_state, LightState::LeftBlinker)) Lights.LeftBlinker = true;
81 if (FLAG_ENABLED(light_state, LightState::Reverse)) Lights.Reverse = true;
82 if (FLAG_ENABLED(light_state, LightState::Fog)) Lights.Fog = true;
83 if (FLAG_ENABLED(light_state, LightState::Interior)) Lights.Interior = true;
84 if (FLAG_ENABLED(light_state, LightState::Special1)) Lights.Special1 = true;
85 if (FLAG_ENABLED(light_state, LightState::Special2)) Lights.Special2 = true;
86 return Lights;
87 }
88
89#endif // LIBCARLA_INCLUDED_FROM_UE4
90
91 /// Returns the current light state as an enum type
92 LightState GetLightStateEnum() const {
93 return static_cast<LightState>(light_state);
94 }
95
96 /// Returns the current light state as a value type
97 flag_type GetLightStateAsValue() const {
98 return light_state;
99 }
100
101 /// Lights state flag, all turned off by default
102 flag_type light_state = static_cast<flag_type>(LightState::None);
103
104 MSGPACK_DEFINE_ARRAY(light_state)
105
106 };
