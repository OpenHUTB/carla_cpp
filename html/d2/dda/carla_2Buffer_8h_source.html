32 /// 请注意，如果需要更多容量，则会分配一个新的内存块，并
33 /// 删除旧的内存块。这意味着默认情况下，缓冲区只能增长。要释放内存，使用 `clear` 或 `pop`。

35 /// 这是一个仅可移动的类型，设计为按值传递时成本较低。如果缓冲区
36 /// 从 BufferPool 中检索，则在销毁时内存会自动推回到池中。

242 /// 重置缓冲区的大小。如果容量不足，当前内存将被丢弃，并分配一个新的大小为 @a size 的内存块。
243 /// allocated.
244 void reset(size_type size) {
245 if (_capacity < size) {
246 log_debug("allocating buffer of", size, "bytes");
247 _data = std::make_unique<value_type[]>(size);
248 _capacity = size;
249 }
250 _size = size;
251 }

254 void reset(uint64_t size) {

256 throw_exception(std::invalid_argument("message size too big"));
257 }

261 /// 调整缓冲区的大小。如果容量不足，将分配一个新的大小为 @a size 的内存块，并复制数据。
262 void resize(uint64_t size) {
263 if(_capacity < size) {
264 std::unique_ptr<value_type[]> data = std::move(_data);
265 uint64_t old_size = size;
266 reset(size);
267 copy_from(data.get(), static_cast<size_type>(old_size));
268 }
269 _size = static_cast<size_type>(size);
270 }

272 /// 释放此缓冲区的内容，并将其大小和容量设置为零。
273 std::unique_ptr<value_type[]> pop() noexcept {
274 _size = 0u;
275 _capacity = 0u;
276 return std::move(_data);
277 }

311 /// @copydoc copy_from(size_type, const Buffer &)
312 template <typename T>
313 typename std::enable_if<boost::asio::is_const_buffer_sequence<T>::value>::type
314 copy_from(size_type offset, const T &source) {
315 reset(boost::asio::buffer_size(source) + offset);
316 DEBUG_ASSERT(boost::asio::buffer_size(source) == size() - offset);
317 DEBUG_ONLY(auto bytes_copied = )
318 boost::asio::buffer_copy(buffer() + offset, source);
319 DEBUG_ASSERT(bytes_copied == size() - offset);
320 }

325 copy_from(size_type offset, const T &source) {
326 copy_from(offset, boost::asio::buffer(source));
327 }
