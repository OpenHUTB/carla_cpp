25 static bool GetControlIsSticky(const AttributesT &attributes) {
26 for (auto &&attribute : attributes) {
27 if (attribute.GetId() == "sticky_control") {
28 return attribute.template As<bool>();
29 }
30 }
31 return true;
32 }
