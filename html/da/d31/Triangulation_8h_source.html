277/// Given a grid cube and an isolevel the triangles (5 max)
278/// required to represent the isosurface in the cube are computed.
279void MeshReconstruction::Triangulate(
280 IntersectInfo const &intersect,
281 Fun3v const &grad,
282 Mesh &mesh)
283{
284 // Cube is entirely in/out of the surface. Generate no triangles.
285 if (intersect.signConfig == 0 || intersect.signConfig == 255)
286 return;
287
288 auto const &tri = signConfigToTriangles[intersect.signConfig];
289
290 for (auto i = 0; tri[i] != -1; i += 3)
291 {
292 auto const &v0 = intersect.edgeVertIndices[tri[i]];
293 auto const &v1 = intersect.edgeVertIndices[tri[i + 1]];
294 auto const &v2 = intersect.edgeVertIndices[tri[i + 2]];
295
296 mesh.vertices.push_back(v0);
297 mesh.vertices.push_back(v1);
298 mesh.vertices.push_back(v2);
299
300 auto normal0 = grad(v0).Normalized();
301 auto normal1 = grad(v1).Normalized();
302 auto normal2 = grad(v2).Normalized();
303
304 mesh.vertexNormals.push_back(normal0);
305 mesh.vertexNormals.push_back(normal1);
306 mesh.vertexNormals.push_back(normal2);
307
308 auto last = static_cast<int>(mesh.vertices.size() - 1);
309
310 mesh.triangles.push_back({last - 2, last - 1, last});
311 }
