458 /// 根据端口号获取交通管理器实例。
459 /// @param port 端口号。
460 /// @return 对应的交通管理器实例指针；如果未找到，则返回nullptr。
461 TrafficManagerBase* GetTM(uint16_t port) const {
462 std::lock_guard<std::mutex> lock(_mutex);
463 auto it = _tm_map.find(port);
464 if (it != _tm_map.end()) {
465 return it->second;
466 }
467 return nullptr;
468 }
