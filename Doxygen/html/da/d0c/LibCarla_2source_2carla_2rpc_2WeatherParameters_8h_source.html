147 bool operator!=(const WeatherParameters &rhs) const {
148 return
149 cloudiness != rhs.cloudiness ||
150 precipitation != rhs.precipitation ||
151 precipitation_deposits != rhs.precipitation_deposits ||
152 wind_intensity != rhs.wind_intensity ||
153 sun_azimuth_angle != rhs.sun_azimuth_angle ||
154 sun_altitude_angle != rhs.sun_altitude_angle ||
155 fog_density != rhs.fog_density ||
156 fog_distance != rhs.fog_distance ||
157 fog_falloff != rhs.fog_falloff ||
158 wetness != rhs.wetness ||
159 scattering_intensity != rhs.scattering_intensity ||
160 mie_scattering_scale != rhs.mie_scattering_scale ||
161 rayleigh_scattering_scale != rhs.rayleigh_scattering_scale ||
162 dust_storm != rhs.dust_storm;
163 }

166 return !(*this != rhs);
