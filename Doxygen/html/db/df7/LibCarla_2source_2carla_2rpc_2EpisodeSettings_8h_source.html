20namespace carla {
21namespace rpc {
22
23 class EpisodeSettings {
24 public:
25 // =========================================================================
26 // -- 公开的数据成员 --------------------------------------------------------
27 // =========================================================================
28
29 // 是否启用同步模式
30 bool synchronous_mode = false;
31
32 // 是否启用无渲染模式
33 bool no_rendering_mode = false;
34
35 // 可选的固定时间间隔，使用 boost::optional 来表示该值可能存在也可能不存在
36 boost::optional<double> fixed_delta_seconds;
37
38 // 是否启用子步，默认为 true
39 bool substepping = true;
40
41 // 两个物理子步之间的最大时间
42 double max_substep_delta_time = 0.01;
43
44 // 一个仿真帧包含的最多物理子步数
45 int max_substeps = 10;
46
47 // 距离剔除：当对象尺寸足够小，距离镜头足够远时，可以不进行渲染，以提高性能
48 // https://www.jianshu.com/p/537d297bedfa
49 float max_culling_distance = 0.0f;
50
51 // 确定性布娃娃系统，默认为 true
52 bool deterministic_ragdolls = true;
53
54 // 该关卡中 距离玩家 3km 内的地图瓦片会被加载
55 float tile_stream_distance = 3000.f; // 3km
56
57 // 距离玩家 2km 内的参与者会被激活
58 float actor_active_distance = 2000.f; // 2km
59
60 // 是否将观众视为自我，默认为 true
61 bool spectator_as_ego = true;
62
63 // 使用 MSGPACK_DEFINE_ARRAY 宏将类的成员变量按顺序打包到 msgpack 中，用于序列化
64 MSGPACK_DEFINE_ARRAY(synchronous_mode, no_rendering_mode, fixed_delta_seconds, substepping,
65 max_substep_delta_time, max_substeps, max_culling_distance, deterministic_ragdolls,
66 tile_stream_distance, actor_active_distance, spectator_as_ego);
67
68 // =========================================================================
69 // -- 构造函数 --------------------------------------------------------------
70 // =========================================================================
71
72 // 默认构造函数，使用编译器生成的默认实现
73 EpisodeSettings() = default;
74
75 // 带有参数的构造函数，可以指定各项设置
76 EpisodeSettings(
77 bool synchronous_mode,
78 bool no_rendering_mode,
79 double fixed_delta_seconds = 0.0,
80 bool substepping = true,
81 double max_substep_delta_time = 0.01,
82 int max_substeps = 10,
83 float max_culling_distance = 0.0f,
84 bool deterministic_ragdolls = true,
85 float tile_stream_distance = 3000.f,
86 float actor_active_distance = 2000.f,
87 bool spectator_as_ego = true)
88 : synchronous_mode(synchronous_mode),
89 no_rendering_mode(no_rendering_mode),
90 fixed_delta_seconds(
91 fixed_delta_seconds > 0.0? fixed_delta_seconds : boost::optional<double>{}),
92 substepping(substepping),
93 max_substep_delta_time(max_substep_delta_time),
94 max_substeps(max_substeps),
95 max_culling_distance(max_culling_distance),
96 deterministic_ragdolls(deterministic_ragdolls),
97 tile_stream_distance(tile_stream_distance),
98 actor_active_distance(actor_active_distance),
99 spectator_as_ego(spectator_as_ego) {}
100
101 // =========================================================================
102 // -- 比较操作符 ------------------------------------------------------------
103 // =========================================================================
104
105 // 重载 == 操作符，用于比较两个 EpisodeSettings 对象是否相等
106 bool operator==(const EpisodeSettings &rhs) const {
107 return
108 (synchronous_mode == rhs.synchronous_mode) &&
109 (no_rendering_mode == rhs.no_rendering_mode) &&
110 (substepping == rhs.substepping) &&
111 (fixed_delta_seconds == rhs.fixed_delta_seconds) &&
112 (max_substep_delta_time == rhs.max_substep_delta_time) &&
113 (max_substeps == rhs.max_substeps) &&
114 (max_culling_distance == rhs.max_culling_distance) &&
115 (deterministic_ragdolls == rhs.deterministic_ragdolls) &&
116 (tile_stream_distance == rhs.tile_stream_distance) &&
117 (actor_active_distance == rhs.actor_active_distance) &&
118 (spectator_as_ego == rhs.spectator_as_ego);
119 }
120
121 // 重载!= 操作符，使用 == 操作符的结果取反
122 bool operator!=(const EpisodeSettings &rhs) const {
123 return!(*this == rhs);
124 }
125
126 // =========================================================================
127 // -- 转换成虚幻4类型 -------------------------------------------------------
128 // =========================================================================
129
130#ifdef LIBCARLA_INCLUDED_FROM_UE4
131 // 从虚幻4的 FEpisodeSettings 类型构造 EpisodeSettings
132 EpisodeSettings(const FEpisodeSettings &Settings)
133 : EpisodeSettings(
134 Settings.bSynchronousMode,
135 Settings.bNoRenderingMode,
136 Settings.FixedDeltaSeconds.Get(0.0),
137 Settings.bSubstepping,
138 Settings.MaxSubstepDeltaTime,
139 Settings.MaxSubsteps,
140 Settings.MaxCullingDistance,
141 Settings.bDeterministicRagdolls,
142 Settings.TileStreamingDistance,
143 Settings.ActorActiveDistance,
144 Settings.SpectatorAsEgo) {
145 constexpr float CMTOM = 1.f/100.f;
146 tile_stream_distance = CMTOM * Settings.TileStreamingDistance;
147 actor_active_distance = CMTOM * Settings.ActorActiveDistance;
148 }
149
150 // 类型转换运算符，将 EpisodeSettings 转换为虚幻4的 FEpisodeSettings 类型
151 operator FEpisodeSettings() const {
152 constexpr float MTOCM = 100.f;
153 FEpisodeSettings Settings;
154 Settings.bSynchronousMode = synchronous_mode;
155 Settings.bNoRenderingMode = no_rendering_mode;
156 if (fixed_delta_seconds.has_value()) {
157 Settings.FixedDeltaSeconds = *fixed_delta_seconds;
158 }
159 Settings.bSubstepping = substepping;
160 Settings.MaxSubstepDeltaTime = max_substep_delta_time;
161 Settings.MaxSubsteps = max_substeps;
162 Settings.MaxCullingDistance = max_culling_distance;
163 Settings.bDeterministicRagdolls = deterministic_ragdolls;
164 Settings.TileStreamingDistance = MTOCM * tile_stream_distance;
165 Settings.ActorActiveDistance = MTOCM * actor_active_distance;
166 Settings.SpectatorAsEgo = spectator_as_ego;
167
168 return Settings;
169 }
170#endif // LIBCARLA_INCLUDED_FROM_UE4
171 };
