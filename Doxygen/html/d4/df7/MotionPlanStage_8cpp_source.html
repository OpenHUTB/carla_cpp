74 const float vehicle_speed_limit = simulation_state.GetSpeedLimit(actor_id); // 通过 simulation_state 对象，按照车辆 ID 获取车辆当前所在位置的速度限制信息（返回浮点数，例如该路段规定的最大行驶速度，单位可能根据模拟场景设定）
75 const Buffer &waypoint_buffer = buffer_map.at(actor_id); // 根据车辆 ID，从 buffer_map 中获取对应的缓冲区数据（Buffer 类型，具体缓冲区的作用可能与车辆的路径规划、临时存储一些周边环境信息等相关，取决于具体实现）

80 current_timestamp = world.GetSnapshot().GetTimestamp(); // 获取当前世界的时间戳信息，通过 world 对象调用 GetSnapshot 函数获取世界的一个快照（可能包含了当前世界的各种状态信息），再从快照中获取时间戳（用于记录当前操作或者决策所处的时间点）
81 StateEntry current_state;// 这里声明了一个 StateEntry 类型的变量 current_state，但后续代码缺失，不清楚具体用途，可能用于记录当前车辆或者整个模拟系统的某种状态信息，等待进一步赋值和使用
82
83 // 实例化传送变换为当前载具变换
84 cg::Transform teleportation_transform = cg::Transform(vehicle_location, vehicle_rotation);
85
86 // 从actor_id状态中获取英雄位置信息
87 cg::Location hero_location = track_traffic.GetHeroLocation();
88 bool is_hero_alive = hero_location != cg::Location(0, 0, 0);
89
90 if (simulation_state.IsDormant(actor_id) && parameters.GetRespawnDormantVehicles() && is_hero_alive) {
91 // 冲洗车辆的控制器状态
92 current_state = {current_timestamp,
93 0.0f, 0.0f,
94 0.0f};
95
96 // 如果表中不存在，则将条目添加到传送持续时间时钟表中
97 if (teleportation_instance.find(actor_id) == teleportation_instance.end()) {
98 teleportation_instance.insert({actor_id, current_timestamp});
99 }
100
101 // 获取传送载具的下限和上限
102 float lower_bound = parameters.GetLowerBoundaryRespawnDormantVehicles();
103 float upper_bound = parameters.GetUpperBoundaryRespawnDormantVehicles();
104 float dilate_factor = (upper_bound-lower_bound)/100.0f;
105
106 // 测量车辆自上次传送以来所经过的时间
107 double elapsed_time = current_timestamp.elapsed_seconds - teleportation_instance.at(actor_id).elapsed_seconds;
108
109 if (parameters.GetSynchronousMode() || elapsed_time > HYBRID_MODE_DT) {
110 float random_sample = (static_cast<float>(random_device.next())*dilate_factor) + lower_bound;
111 NodeList teleport_waypoint_list = local_map->GetWaypointsInDelta(hero_location, ATTEMPTS_TO_TELEPORT, random_sample);
112 if (!teleport_waypoint_list.empty()) {
113 for (auto &teleport_waypoint : teleport_waypoint_list) {
114 GeoGridId geogrid_id = teleport_waypoint->GetGeodesicGridId();
115 if (track_traffic.IsGeoGridFree(geogrid_id)) {
116 teleportation_transform = teleport_waypoint->GetTransform();
117 teleportation_transform.location.z += 0.5f;
118 track_traffic.AddTakenGrid(geogrid_id, actor_id);
119 break;
120 }
121 }
122 }
123 }
124 output_array.at(index) = carla::rpc::Command::ApplyTransform(actor_id, teleportation_transform);
125
126 // 在传送车辆后，使用新的变换更新模拟状态
127 KinematicState kinematic_state{teleportation_transform.location,
128 teleportation_transform.rotation,
129 vehicle_velocity, vehicle_speed_limit,
130 vehicle_physics_enabled, simulation_state.IsDormant(actor_id),
131 teleportation_transform.location};
132 simulation_state.UpdateKinematicState(actor_id, kinematic_state);
133 }
134
135 else {
136
137 // 目标车速
138 float max_target_velocity = parameters.GetVehicleTargetVelocity(actor_id, vehicle_speed_limit) / 3.6f;
139
140 // 接近地标时减速的算法
141 float max_landmark_target_velocity = GetLandmarkTargetVelocity(*(waypoint_buffer.at(0)), vehicle_location, actor_id, max_target_velocity);
142
143 // 转弯处减速算法
144 float max_turn_target_velocity = GetTurnTargetVelocity(waypoint_buffer, max_target_velocity);
145 max_target_velocity = std::min(std::min(max_target_velocity, max_landmark_target_velocity), max_turn_target_velocity);
146
147 // 碰撞处理与目标速度修正
148 std::pair<bool, float> collision_response = CollisionHandling(collision_hazard, tl_hazard, vehicle_velocity,
149 vehicle_heading, max_target_velocity);
150 bool collision_emergency_stop = collision_response.first;
151 float dynamic_target_velocity = collision_response.second;
152
153 // 如果路口后没有足够的空间，请勿进入路口
154 bool safe_after_junction = SafeAfterJunction(localization, tl_hazard, collision_emergency_stop);
155
156 // 遇到碰撞或交通灯危险时
157 bool emergency_stop = tl_hazard || collision_emergency_stop || !safe_after_junction;
158
159 if (vehicle_physics_enabled && !simulation_state.IsDormant(actor_id)) {// 判断车辆的物理模拟是否启用（vehicle_physics_enabled为true表示启用），并且车辆是否处于休眠状态（!simulation_state.IsDormant(actor_id)表示非休眠状态）
160 ActuationSignal actuation_signal{0.0f, 0.0f, 0.0f};// 创建一个ActuationSignal类型的结构体（或类）对象actuation_signal，并初始化为{0.0f, 0.0f, 0.0f}
161
162 const float target_point_distance = std::max(vehicle_speed * TARGET_WAYPOINT_TIME_HORIZON,
163 MIN_TARGET_WAYPOINT_DISTANCE);// 计算目标点距离，取车辆速度乘以TARGET_WAYPOINT_TIME_HORIZON（目标路点时间范围，可能表示预测的未来某个时间段）
164 const SimpleWaypointPtr &target_waypoint = GetTargetWaypoint(waypoint_buffer, target_point_distance).first;// 调用GetTargetWaypoint函数，传入路点缓冲区（waypoint_buffer）和刚计算出的目标点距离（target_point_distance），
165 cg::Location target_location = target_waypoint->GetLocation(); // 获取目标路点的位置信息（cg::Location类型，可能包含三维坐标等位置相关数据），赋值给target_location变量
166
167 float offset = parameters.GetLaneOffset(actor_id); // 获取车辆在车道上的偏移量，通过parameters对象调用GetLaneOffset函数，传入车辆ID（actor_id）获取对应的车道偏移量，
168 auto right_vector = target_waypoint->GetTransform().GetRightVector(); // 获取目标路点的右方向向量（GetRightVector函数返回的可能是表示路点所在位置的右侧方向的三维向量，用于确定横向方向），
169 auto offset_location = cg::Location(cg::Vector3D(offset*right_vector.x, offset*right_vector.y, 0.0f));// 根据车道偏移量和右方向向量计算出偏移后的位置向量，通过将偏移量与右方向向量的各分量相乘构建一个新的三维向量，
170 target_location = target_location + offset_location;// 将之前获取的目标位置（target_location）加上计算出的偏移位置（offset_location），得到考虑车道偏移后的实际目标位置
171 // 这样车辆后续的行驶目标就会根据车道偏移情况进行相应调整
172
173 float dot_product = DeviationDotProduct(vehicle_location, vehicle_heading, target_location);// 调用DeviationDotProduct函数，传入车辆当前位置（vehicle_location）、车辆行驶方向（vehicle_heading）和调整后的目标位置（target_location），
174 float cross_product = DeviationCrossProduct(vehicle_location, vehicle_heading, target_location);// 调用DeviationCrossProduct函数，传入同样的三个参数，计算它们之间的叉积（叉积可以得到一个与两个输入向量都垂直的向量，其大小等信息在某些场景下可用于判断方向关系等），
175 dot_product = acos(dot_product) / PI; // 对计算出的点积结果进行处理，通过调用acos函数（反余弦函数）将点积转换为对应的夹角弧度值（因为点积与向量夹角余弦值相关）
176 if (cross_product < 0.0f) {
177// 如果叉积小于0.0f，说明车辆相对目标位置在某个特定的方向关系上（可能是根据叉积正负判断的左右方向等情况，具体取决于向量定义和坐标系设定）
178 // 则将点积乘以 -1.0f，可能是为了统一方向偏差的表示方式，使得点积值能更准确地反映符合预期逻辑的角度偏差情况
179 dot_product *= -1.0f;
180 }
181 const float angular_deviation = dot_product; // 将处理后的点积值赋值给angular_deviation变量，从变量名推测它表示车辆与目标位置之间的角度偏差
182 const float velocity_deviation = (dynamic_target_velocity - vehicle_speed) / dynamic_target_velocity;
183// 计算速度偏差，用动态目标速度（dynamic_target_velocity，可能是根据路况、规划等因素设定的车辆期望达到的目标速度）减去车辆当前速度（vehicle_speed）
184 // 如果未找到车辆的上一个状态，则初始化状态条目
185 if (pid_state_map.find(actor_id) == pid_state_map.end()) {
186 const auto initial_state = StateEntry{current_timestamp, 0.0f, 0.0f, 0.0f};
187 pid_state_map.insert({actor_id, initial_state});
188 }
189
190 // 检索先前状态
191 traffic_manager::StateEntry previous_state;
192 previous_state = pid_state_map.at(actor_id);
193
194 // 选择PID参数
195 std::vector<float> longitudinal_parameters;
196 std::vector<float> lateral_parameters;
197 if (vehicle_speed > HIGHWAY_SPEED) {
198 longitudinal_parameters = highway_longitudinal_parameters;
199 lateral_parameters = highway_lateral_parameters;
200 } else {
201 longitudinal_parameters = urban_longitudinal_parameters;
202 lateral_parameters = urban_lateral_parameters;
203 }
204
205 //如果为车辆启用了物理效果，请使用PID控制器
206 // 车辆状态更新
207 current_state = {current_timestamp, angular_deviation, velocity_deviation, 0.0f};
208
209 // 控制器驱动
210 actuation_signal = PID::RunStep(current_state, previous_state,
211 longitudinal_parameters, lateral_parameters);
212
213 if (emergency_stop) {
214 actuation_signal.throttle = 0.0f;
215 actuation_signal.brake = 1.0f;
216 }
217
218 // 构建执行信号
219
220 carla::rpc::VehicleControl vehicle_control;
221 vehicle_control.throttle = actuation_signal.throttle;
222 vehicle_control.brake = actuation_signal.brake;
223 vehicle_control.steer = actuation_signal.steer;
224
225 output_array.at(index) = carla::rpc::Command::ApplyVehicleControl(actor_id, vehicle_control);
226
227 // 更新PID状态
228 current_state.steer = actuation_signal.steer;
229 StateEntry &state = pid_state_map.at(actor_id);
230 state = current_state;
231 }
232 // 对于无物理特性的载具，确定传送时的位置和方向
233 else {
234 // 冲洗车辆的控制器状态
235 current_state = {current_timestamp,
236 0.0f, 0.0f,
237 0.0f};
238
239 // 如果不在表中，则将条目添加到传送持续时间时钟表中
240 if (teleportation_instance.find(actor_id) == teleportation_instance.end()) {
241 teleportation_instance.insert({actor_id, current_timestamp});
242 }
243
244 // 测量车辆自上次传送以来的时间
245 double elapsed_time = current_timestamp.elapsed_seconds - teleportation_instance.at(actor_id).elapsed_seconds;
246
247 // 在车辆前方找到一个传送位置，以实现预期的速度
248 if (!emergency_stop && (parameters.GetSynchronousMode() || elapsed_time > HYBRID_MODE_DT)) {
249
250 // 目标位移量以达到目标速度
251 const float target_displacement = dynamic_target_velocity * HYBRID_MODE_DT_FL;
252 SimpleWaypointPtr teleport_target = waypoint_buffer.front();
253 cg::Transform target_base_transform = teleport_target->GetTransform();
254 cg::Location target_base_location = target_base_transform.location;
255 cg::Vector3D target_heading = target_base_transform.GetForwardVector();
256 cg::Vector3D correct_heading = (target_base_location - vehicle_location).MakeSafeUnitVector(EPSILON);
257
258 if (vehicle_location.Distance(target_base_location) < target_displacement) {
259 cg::Location teleportation_location = vehicle_location + cg::Location(target_heading.MakeSafeUnitVector(EPSILON) * target_displacement);
260 teleportation_transform = cg::Transform(teleportation_location, target_base_transform.rotation);
261 }
262 else {
263 cg::Location teleportation_location = vehicle_location + cg::Location(correct_heading * target_displacement);
264 teleportation_transform = cg::Transform(teleportation_location, target_base_transform.rotation);
265 }
266 // 在紧急停止的情况下，请保持在同一位置
267 // 此外，在异步模式下，每 dt 时间仅传送一次
268 } else {
269 teleportation_transform = cg::Transform(vehicle_location, simulation_state.GetRotation(actor_id));
270 }
271 // 构建执行信号
272 output_array.at(index) = carla::rpc::Command::ApplyTransform(actor_id, teleportation_transform);
273 simulation_state.UpdateKinematicHybridEndLocation(actor_id, teleportation_transform.location);
274 }
275 }

278bool MotionPlanStage::SafeAfterJunction(const LocalizationData &localization,
279 const bool tl_hazard,
280 const bool collision_emergency_stop) {// MotionPlanStage类中的成员函数SafeAfterJunction，用于判断车辆在经过路口后是否处于安全状态
281
282 SimpleWaypointPtr junction_end_point = localization.junction_end_point;
283// 获取LocalizationData中的路口端点指针（SimpleWaypointPtr类型，可能指向表示路口结束位置的路点对象，用于后续位置相关的判断和计算）
284 SimpleWaypointPtr safe_point = localization.safe_point;// 获取LocalizationData中的安全点指针（SimpleWaypointPtr类型，可能是经过路口后被认定为安全的一个位置对应的路点对象）
285

369 }
370 }

374 // 不要每帧减速超过PERC_MAX_SLOWDOWN
375 dynamic_target_velocity = vehicle_speed - max_gradual_velocity;
376 }
377 dynamic_target_velocity = std::min(max_target_velocity, dynamic_target_velocity);
378
379 return {collision_emergency_stop, dynamic_target_velocity};
