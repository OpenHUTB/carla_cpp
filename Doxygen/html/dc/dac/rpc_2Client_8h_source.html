27 // 定义了一个名为Client的类，从类的功能来看，它主要是作为与远程服务进行交互的客户端角色，
28 // 对底层的RPC客户端功能进行了一定的封装，以便在Carla项目中更方便地使用。
29 class Client {
30 public:
31 // 这是一个模板构造函数，接受任意数量的参数（Args &&... args），
32 // 通过使用 std::forward<Args>(args) 进行完美转发，将传入的参数原封不动地传递给成员变量 _client 的构造函数，
33 // 避免了不必要的拷贝操作，保证了参数的右值引用特性得以保留（如果传入的是右值的话），从而提高效率。
34 // 它的作用是初始化这个Client类的实例，同时初始化其内部的 _client 成员变量。
35 template <typename... Args>
36 explicit Client(Args &&... args)
37 : _client(std::forward<Args>(args)...) {}
38
39 // 该方法用于设置客户端与远程服务交互时的超时时间，
40 // 它调用了底层的 ::rpc::client 对象（即成员变量 _client）的 set_timeout 方法来实现具体的设置操作，
41 // 参数 value 就是要设置的超时时间值（类型为int64_t）。
42 void set_timeout(int64_t value) {
43 _client.set_timeout(value);
44 }
45
46 // 这个方法用于获取当前客户端与远程服务交互时设置的超时时间，
47 // 通过调用底层 ::rpc::client 对象（_client）的 get_timeout 方法来获取超时时间值，
48 // 并将获取到的值返回给调用者，返回值类型由底层的 get_timeout 方法决定。
49 auto get_timeout() const {
50 return _client.get_timeout();
51 }
52
53 // 这是一个模板函数，用于发起一个同步的远程过程调用（RPC）。
54 // 它接受一个函数名（function，类型为const std::string &，表示对远程服务中某个具体函数的调用），
55 // 以及任意数量的其他参数（Args &&... args），这些参数会被转发给底层的RPC调用机制。
56 // 在调用底层 _client 的 call 方法时，除了传入函数名和转发的参数外，还传入了 Metadata::MakeSync()，
57 // 推测是告知底层此次调用是同步的，并传递一些相关的元数据信息用于控制调用过程，
58 // 最后返回远程调用的结果，结果类型由底层的RPC调用返回类型决定。
59 template <typename... Args>
60 auto call(const std::string &function, Args &&... args) {
61 return _client.call(function, Metadata::MakeSync(), std::forward<Args>(args)...);
62 }
63
64 // 此方法用于发起一个异步的远程过程调用（RPC）。
65 // 同样接受一个函数名（function，类型为const std::string &）以及任意数量的其他参数（Args &&... args），
66 // 调用底层 _client 的 async_call 方法时，传入函数名、Metadata::MakeAsync()（推测是用于告知底层此次调用是异步的，同时传递相关元数据）
67 // 和转发的参数，以此实现异步调用的发起，而该方法本身无返回值，因为异步调用结果通常需要通过其他方式（比如回调函数等）来获取。
68 template <typename... Args>
69 void async_call(const std::string &function, Args &&... args) {
70 _client.async_call(function, Metadata::MakeAsync(), std::forward<Args>(args)...);
71 }
72
73 private:
74 // 定义了一个私有成员变量 _client，类型为 ::rpc::client，
75 // 它是底层实际用于和远程服务进行交互的RPC客户端对象，
76 // 本类（Client）中的各种方法基本都是围绕对这个底层客户端对象的操作来实现与远程服务通信功能的。
77 ::rpc::client _client;
78 };
79
80 } // namespace rpc
81} // namespace carla

84// 使得客户端和远程服务之间的通信变得简单且高效，特别适合应用在像Carla这样的自动驾驶模拟器中，
85// 方便模拟器不同模块之间通过RPC机制进行交互通信，例如控制模拟车辆、获取环境数据等场景下使用。
86// This design makes the communication between the client and the remote service simple and efficient,
87// and is suitable for use in an autonomous driving simulator like Carla.
