24 class ProfilerData { // 性能数据类
25 public:
26
27 explicit ProfilerData(std::string name, bool print_fps = false) // 构造函数，初始化名称和是否打印FPS
28 : _name(std::move(name)), // 移动名称
29 _print_fps(print_fps) {} // 设置打印FPS布尔值
30
31 ~ProfilerData(); // 析构函数声明
32
33 void Annotate(const StopWatch &stop_watch) { // 注解方法，记录时间数据
34 const auto elapsed_microseconds = stop_watch.GetElapsedTime<std::chrono::microseconds>(); // 获取经过的微秒数
35 ++_count; // 计数增加
36 _total_microseconds += elapsed_microseconds; // 总时间增加
37 _max_elapsed = std::max(elapsed_microseconds, _max_elapsed); // 更新最大时间
38 _min_elapsed = std::min(elapsed_microseconds, _min_elapsed); // 更新最小时间
39 }
40
41 float average() const { // 计算平均时间
42 return ms(_total_microseconds) / static_cast<float>(_count); // 将总时间（微秒）转换为毫秒，然后除以事件发生的次数，得到平均每次事件的毫秒数
43 }
44
45 float maximum() const { // 获取最大时间
46 return ms(_max_elapsed); // 返回最大时间（毫秒）
47 }
48
49 float minimum() const { // 获取最小时间
50 return ms(_min_elapsed); // 返回最小时间（毫秒）
51 }
52
53 private:
54
55 static inline float ms(size_t microseconds) { // 将微秒转换为毫秒
56 return 1e-3f * static_cast<float>(microseconds); // 转换公式
57 }
58
59 static inline float fps(float milliseconds) { // 根据毫秒计算FPS
60 return milliseconds > 0.0f ? (1e3f / milliseconds) : std::numeric_limits<float>::max(); // 如果 milliseconds 大于 0，则返回其倒数（每秒），否则返回无穷大
61 }
62
63 const std::string _name; // 性能监测名称
64
65 const bool _print_fps; // 是否打印FPS
66
67 size_t _count = 0u; // 记录次数
68
69 size_t _total_microseconds = 0u; // 总微秒数
70
71 size_t _max_elapsed = 0u; // 最大经历的时间
72
73 size_t _min_elapsed = std::numeric_limits<size_t>::max(); // 最小经历的时间（初始化为最大值）
74 };
