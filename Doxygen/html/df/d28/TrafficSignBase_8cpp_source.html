29// 定义一个函数GetTriggerVolumes，用于获取交通标志的触发体积组件数组
30// 函数返回值类型为TArray<UBoxComponent*>，即一个UBoxComponent类型的数组
31TArray<UBoxComponent*> ATrafficSignBase::GetTriggerVolumes() const
32{
33 // 创建一个USignComponent类型的数组，用于存储获取到的标志组件
34 TArray<USignComponent*> Components;
35 // 通过GetComponents函数获取该交通标志Actor下的所有USignComponent组件
36 // 第二个参数false表示不包括已经被标记为待删除的组件
37 GetComponents<USignComponent>(Components, false);
38 // 判断获取到的标志组件数组的元素数量是否大于0
39 if (Components.Num())
40 {
41 // 如果有标志组件，取出数组中的第一个标志组件
42 USignComponent* SignComponent = Components[0];
43 // 调用该标志组件的GetEffectTriggerVolume函数获取其触发体积组件数组并返回
44 return SignComponent->GetEffectTriggerVolume();
45 }
46 else
47 {
48 // 如果没有获取到标志组件，创建一个新的UBoxComponent类型的数组，用于存储触发体积组件
49 TArray<UBoxComponent*> TriggerVolumes;
50 // 将该交通标志Actor自身的触发体积组件（通过GetTriggerVolume函数获取）添加到数组中
51 TriggerVolumes.Add(GetTriggerVolume());
52 // 返回包含该交通标志Actor自身触发体积组件的数组
53 return TriggerVolumes;
54 }
55}
